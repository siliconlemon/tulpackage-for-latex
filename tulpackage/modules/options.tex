% Přepínač stylu podle názvu fakulty/ústavu
\newcommand{\switchTULstyle}[1]{%
    \renewcommand{\sekce@TUL}{#1}%
    \renewcommand{\tulcolor}{tul#1}%
    \renewcommand{\titlecolor}{tul#1}%
    \renewcommand{\CZlogo}{\csname logo#1\endcsname}%
    \renewcommand{\CZlogowh}{\csname logo#1wh\endcsname}%
    \renewcommand{\ENlogo}{\csname ENlogo#1\endcsname}%
    \renewcommand{\ENlogowh}{\csname ENlogo#1wh\endcsname}%
    \renewcommand{\CZlogotext}{\csname logo#1text\endcsname}%
    \renewcommand{\CZlogotextwh}{\csname logo#1textwh\endcsname}%
    \renewcommand{\ENlogotext}{\csname ENlogo#1text\endcsname}%
    \renewcommand{\ENlogotextwh}{\csname ENlogo#1textwh\endcsname}%
    \renewcommand{\symbolTUL}{\csname symbol#1\endcsname}%
}%

% Možnost přepnutí popisků pro obrázky, tabulky a bloky kódu do kurzivy

% Zarovnání popisků doleva (Figures, Tables, Listings)
\DeclareOption{captionsleft}{%
    \setboolean{TUL@figcapsleft}{true}%
    \setboolean{TUL@tabcapsleft}{true}%
    \setboolean{TUL@lstcapsleft}{true}%
    \setboolean{TUL@srccapsleft}{true}%
}%
\DeclareOption{figcapsleft}{\setboolean{TUL@figcapsleft}{true}}%
\DeclareOption{tabcapsleft}{\setboolean{TUL@tabcapsleft}{true}}%
\DeclareOption{lstcapsleft}{\setboolean{TUL@lstcapsleft}{true}}%
\DeclareOption{srccapsleft}{\setboolean{TUL@srccapsleft}{true}}%

% Volby zarovnání plovoucích objektů doleva
% POZNÁMKA: Pokud TRUE, zarovnávají automaticky i popisky na levou stranu
% POZNÁMKA: Po změně těchto nastavení může být potřeba kompilovat dvakrát za sebou
% POZNÁMKA: TUL@listingsleft momentálně neovlivňuje vůbec nic, mění se zde jen
% zarovnání popisků - kód nemá smysl centrovat
\DeclareOption{figuresleft}{%
    \setboolean{TUL@figuresleft}{true}%
    \setboolean{TUL@figcapsleft}{true}%
}%
\DeclareOption{tablesleft}{%
    \setboolean{TUL@tablesleft}{true}%
    \setboolean{TUL@tabcapsleft}{true}%
}%
\DeclareOption{listingsleft}{%
    \setboolean{TUL@listingsleft}{true}%
    \setboolean{TUL@lstcapsleft}{true}%
}%

% Meta volba zarovnání ÚPLNĚ VŠEHO doleva (ilustrace, popisky, zdrojové titulky)
\DeclareOption{everythingleft}{%
    \setboolean{TUL@figuresleft}{true}%
    \setboolean{TUL@tablesleft}{true}%
    \setboolean{TUL@listingsleft}{true}%
    \setboolean{TUL@figcapsleft}{true}%
    \setboolean{TUL@tabcapsleft}{true}%
    \setboolean{TUL@lstcapsleft}{true}%
    \setboolean{TUL@srccapsleft}{true}%
}%
